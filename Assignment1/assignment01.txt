function Evaluator(){
    var sum = 0;
    var average = 0;
    for (var i=0; i<arguments.length; i++){
        sum += arguments[i];
    }
    average = sum / arguments.length;
    
    if (average >= 50){
        return true;
    } else {
        return false;
    }
}
console.log("Average, greater than or equal to 50: " + Evaluator(20,30,1));

var grader = function (numScore){
  var mark;
  if (numScore >= 90)
    mark = 'A';
  else if (numScore >= 80)
    mark = 'B';
  else if (numScore >= 70)
    mark = 'c';
  else if (numScore >= 60)
    mark = 'D';
  else if (numScore >= 50)
    mark = 'F';
  
  return mark;
}

console.log(grader(95));
console.log(grader(85));
console.log(grader(75));
